import 'package:flutter/material.dart';
import 'package:google_maps_flutter/google_maps_flutter.dart';

void main() {
  runApp(GoogleMap1());
}

class GoogleMap1 extends StatelessWidget {
  const GoogleMap1({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(debugShowCheckedModeBanner: false, home: TestMap());
  }
}

class TestMap extends StatefulWidget {
  const TestMap({super.key});

  @override
  State<TestMap> createState() => _TestMapState();
}

class _TestMapState extends State<TestMap> {
  late GoogleMapController mapShow;
  LatLng point = LatLng(16.187, 103.3045);

  void createGoogleMap(GoogleMapController map) {
    mapShow = map;
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Test Google Map'),
        centerTitle: true,
        backgroundColor: const Color.fromARGB(255, 1, 131, 62),
      ),
      body: GoogleMap(
        onMapCreated: createGoogleMap,
        initialCameraPosition: CameraPosition(
          target: point,
          zoom: 10.0
          ),
      )
    );
  }
}
